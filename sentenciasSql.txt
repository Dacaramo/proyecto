CREATE TABLE aspirante(
    id INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(40) NOT NULL,
    clave VARCHAR(40) NOT NULL,
    edad INT,
    PRIMARY KEY(id)
);

CREATE TABLE empleador(
    id INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(40) NOT NULL,
    clave VARCHAR(40) NOT NULL,
    PRIMARY KEY(id)
);

CREATE TABLE oferta(
    id INT NOT NULL AUTO_INCREMENT,
    nombre VARCHAR(40) NOT NULL,
    sector VARCHAR(40) NOT NULL,
    edad_requerida INT NOT NULL,
    sueldo DECIMAL(11,3) NOT NULL,
    id_empleador INT NOT NULL,
    PRIMARY KEY(id),
    CONSTRAINT FK_oferta FOREIGN KEY(id_empleador) REFERENCES empleador(id) ON DELETE CASCADE ON UPDATE CASCADE  
);

CREATE TABLE ofertas_x_aspirante(
    id_oferta INT NOT NULL,
    id_aspirante INT NOT NULL,
    PRIMARY KEY(id_oferta, id_aspirante),
    CONSTRAINT FK_oferta_x_aspirante_1 FOREIGN KEY(id_oferta) REFERENCES oferta(id) ON DELETE CASCADE ON UPDATE CASCADE,
    CONSTRAINT FK_oferta_x_aspirante_2 FOREIGN KEY(id_aspirante) REFERENCES aspirante(id) ON DELETE CASCADE ON UPDATE CASCADE  
);
